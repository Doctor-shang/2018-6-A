        //////////////////////////////////////////////////////////////////////////
        var scope = "global scope";

        function checkscope() {
            var scope = "local scope";

            function f() {
                //f在这里定义,这里输出local scope,那么不管f什么时候在什么地方被调用,都会输出local scope
                console.log(scope);
            }
            return f();
        }
        checkscope();
        //////////////////////////////////////////////////////////////////////////
        var scope_ = "global scope";

        function checkscope_() {
            var scope_ = "local scope";

            function f() {
                //f在这里定义,这里输出local scope,那么不管f什么时候在什么地方被调用,都会输出local scope
                console.log(scope_);
            }
            return f;
        }
        checkscope_()();
        //return f代表返回函数本身
        //return f()代表返回函数的执行结果
        //在javascript中甭管函数在哪里调用,输出的值一定是函数定义的地方输出的值
